#include <stdio.h>

int main() {
    int primeArr = 11;
    int prime = 0;
    
    for(int i = 2; i <= primeArr / 2; i++)
    {
        if((primeArr % i) == 0)
        {
            prime = 1;
            break; // no need to continue checking if it's not prime
        }
    }
    
    if(prime == 0)
    {
        printf("%d is a prime number\n", primeArr);
    }
    else
    {
        printf("%d is not a prime number\n", primeArr);
    }
    
    return 0;
}
